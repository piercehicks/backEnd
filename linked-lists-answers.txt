//Question 1
// A.) explanation of the bug: current equals the next node when it should be null
//B.) functional implementation of the method: See below

LinkedList.prototype.push = function(val){  
    let node = {
       value: val,
       next: null
    }

    if(!this.head){
      this.head = node;
      this.length += 1;
      return node;
    }
    else{
      current = this.head;

      while(current.next){
        current = current.next;
      }

      current.next = node;
      this.length += 1;
      return node;
    }
  }
